from scapy.all import sniff, IP, TCP, UDP, ICMP
from datetime import datetime

def packet_callback(packet):
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    
    if IP in packet:
        ip_src = packet[IP].src
        ip_dst = packet[IP].dst
        proto = packet[IP].proto
        
        print(f"\n[{timestamp}] New Packet:")
        print(f"Source IP: {ip_src}")
        print(f"Destination IP: {ip_dst}")
        
        if TCP in packet:
            print(f"Protocol: TCP")
            print(f"Source Port: {packet[TCP].sport}")
            print(f"Destination Port: {packet[TCP].dport}")
        elif UDP in packet:
            print(f"Protocol: UDP")
            print(f"Source Port: {packet[UDP].sport}")
            print(f"Destination Port: {packet[UDP].dport}")
        elif ICMP in packet:
            print(f"Protocol: ICMP")
        else:
            print(f"Protocol: {proto}")
        
        print("-" * 50)

def main():
    print("Starting Network Packet Sniffer...")
    print("Press Ctrl+C to stop\n")
    print("Note: This script requires admin/root privileges to run")
    print("Install scapy: pip install scapy\n")
    
    sniff(prn=packet_callback, count=20)

if __name__ == "__main__":
    main()
